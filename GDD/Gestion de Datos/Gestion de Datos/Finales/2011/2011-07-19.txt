CREATE TRIGGER trg_sexo
ON persona
AFTER INSERT OR UPDATE
FOR EACH ROW

BEGIN
  IF(:new.sexo = 'F' AND EXISTS(SELECT * FROM persona WHERE padre = :new.id)
    RAISE ERROR
  ELSIF(:new.sexo = 'M' AND EXISTS(SELECT * FROM persona WHERE madre = :new.id)
    RAISE ERROR
END;
